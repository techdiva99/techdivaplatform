src/components/common/TechDivaLogo.jsx
import React from 'react';
import logoSvg from '../../assets/images/techdiva-logo.svg';

const TechDivaLogo = ({ className = "", size = "large", style = {} }) => {
  const dimensions = size === "large" ? 200 : 100;
  
  const sizeMap = {
    small: 100,
    medium: 150,
    large: 200,
    xlarge: 300
  };
  
  const logoSize = sizeMap[size] || dimensions;
  
  return (
    <img 
      src={logoSvg}
      alt="TechDIVA - AI companion for women in tech"
      className={className}
      width={logoSize}
      height={logoSize}
      style={{
        display: 'inline-block',
        ...style
      }}
    />
  );
};

export default TechDivaLogo;
Alternative: With error handling and fallback
import React, { useState } from 'react';
import logoSvg from '../../assets/images/techdiva-logo.svg';

const TechDivaLogo = ({ className = "", size = "large", style = {} }) => {
  const [imgError, setImgError] = useState(false);
  
  const sizeMap = {
    small: 100,
    medium: 150,
    large: 200,
    xlarge: 300
  };
  
  const logoSize = sizeMap[size] || 200;
  
  // Fallback component if image fails to load
  if (imgError) {
    return (
      <div 
        className={className}
        style={{ 
          width: logoSize, 
          height: logoSize, 
          backgroundColor: '#ffecf8',
          display: 'flex',
          alignItems: 'center',
          justifyContent: 'center',
          borderRadius: '50%',
          border: '2px solid #ff67c7',
          ...style
        }}
      >
        <span style={{ 
          color: '#ff67c7', 
          fontWeight: 'bold',
          fontSize: logoSize / 5
        }}>
          TD
        </span>
      </div>
    );
  }
  
  return (
    <img 
      src={logoSvg}
      alt="TechDIVA - AI companion for women in tech"
      className={className}
      width={logoSize}
      height={logoSize}
      style={{
        display: 'inline-block',
        ...style
      }}
      onError={() => setImgError(true)}
      loading="lazy"
    />
  );
};

export default TechDivaLogo;
Alternative: With responsive sizing
import React from 'react';
import logoSvg from '../../assets/images/techdiva-logo.svg';

const TechDivaLogo = ({ 
  className = "", 
  size = "large",
  responsive = false,
  style = {} 
}) => {
  const sizeMap = {
    small: { width: 100, height: 100 },
    medium: { width: 150, height: 150 },
    large: { width: 200, height: 200 },
    xlarge: { width: 300, height: 300 }
  };
  
  const dimensions = sizeMap[size] || sizeMap.large;
  
  const imageStyles = responsive 
    ? { 
        maxWidth: '100%', 
        height: 'auto',
        width: `${dimensions.width}px`,
        ...style 
      }
    : { 
        width: `${dimensions.width}px`,
        height: `${dimensions.height}px`,
        ...style 
      };
  
  return (
    <img 
      src={logoSvg}
      alt="TechDIVA - AI companion for women in tech"
      className={className}
      style={imageStyles}
    />
  );
};

export default TechDivaLogo;
